25.5. Integrasi Dengan Holders Calculation
Gunakan hasil holders breakdown (whale/large/medium/small) dari backend atau API internal.
Jika belum ada, buat fungsi:
function classifyHolders(balances, totalSupply) {
  let whale = 0, large = 0, medium = 0, small = 0;

  balances.forEach(b => {
    const pct = b / totalSupply;
    if (pct >= 0.01) whale++;
    else if (pct >= 0.001) large++;
    else if (pct >= 0.0001) medium++;
    else small++;
  });

  return [whale, large, medium, small];
}
